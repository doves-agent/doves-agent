技能开发

技能（Skill）是比工具更高层级的执行逻辑模块。 工具是原子操作，技能是编排多个工具完成复杂任务的执行方案。

工具 vs 技能

对比项 工具（Tool） 技能（Skill）
粒度 原子操作 多步骤流程
调用方 LLM 直接调用 策略层编排调用
复杂度 单一输入输出 可包含条件分支、循环、多工具协同
示例 「读取文件」「查询数据」 「代码审查流程」「数据分析报告生成」

文件结构

技能文件放在扩展目录的 skills/ 子目录下：

我的扩展/
└── skills/
    ├── 数据清洗.js
    ├── 报告生成.js
    └── 批量处理.js

技能定义格式

export default {
  name: '数据清洗',
  description: '对导入的原始数据执行清洗流程：去重、格式标准化、异常值处理',

  // 技能需要的输入参数
  parameters: {
    type: 'object',
    properties: {
      数据源: {
        type: 'string',
        description: '待清洗的数据集合名称'
      },
      规则: {
        type: 'object',
        description: '清洗规则配置',
        properties: {
          去重字段: { type: 'array', items: { type: 'string' } },
          必填字段: { type: 'array', items: { type: 'string' } },
          日期格式: { type: 'string', default: 'YYYY-MM-DD' }
        }
      }
    },
    required: ['数据源']
  },

  // 技能依赖的工具列表（文档用途，非强制）
  usesTools: ['查询数据', '更新记录', '删除记录'],

  // 执行逻辑
  execute: async (params, ctx) => {
    const { 数据源, 规则 = {} } = params;
    const { 去重字段 = [], 必填字段 = [], 日期格式 = 'YYYY-MM-DD' } = 规则;

    // 第 1 步：读取原始数据
    ctx.log.info('开始数据清洗', { 数据源 });
    const 原始数据 = await ctx.db.find('user_data', 数据源, {});

    if (原始数据.length === 0) {
      return { success: true, message: '数据集为空，无需清洗', 统计: { 总数: 0 } };
    }

    // 第 2 步：去重
    let 清洗后 = 原始数据;
    let 重复数 = 0;

    if (去重字段.length > 0) {
      const seen = new Set();
      清洗后 = 原始数据.filter(doc => {
        const key = 去重字段.map(f => doc[f]).join('|');
        if (seen.has(key)) {
          重复数++;
          return false;
        }
        seen.add(key);
        return true;
      });
    }

    // 第 3 步：必填字段检查
    let 不完整数 = 0;
    清洗后 = 清洗后.filter(doc => {
      const 完整 = 必填字段.every(f => doc[f] != null && doc[f] !== '');
      if (!完整) 不完整数++;
      return 完整;
    });

    // 第 4 步：写回清洗结果
    if (重复数 > 0 || 不完整数 > 0) {
      // 清空并重写（简化示例，实际应使用事务）
      await ctx.db.deleteMany('user_data', 数据源, {});
      if (清洗后.length > 0) {
        await ctx.db.insertMany('user_data', 数据源, 清洗后);
      }
    }

    const 统计 = {
      原始数量: 原始数据.length,
      清洗后数量: 清洗后.length,
      去除重复: 重复数,
      去除不完整: 不完整数
    };

    ctx.log.info('数据清洗完成', 统计);

    return {
      success: true,
      message: `清洗完成：原始 ${统计.原始数量} 条 → 保留 ${统计.清洗后数量} 条`,
      统计
    };
  }
};

技能类别

技能按功能模式可分为几类：

流水线型

按固定步骤顺序执行，前一步的输出是后一步的输入：

// 流水线型技能结构
execute: async (params, ctx) => {
  const step1Result = await 步骤1(params, ctx);
  const step2Result = await 步骤2(step1Result, ctx);
  const step3Result = await 步骤3(step2Result, ctx);
  return step3Result;
}

条件分支型

根据输入或中间结果选择不同的执行路径：

// 条件分支型技能结构
execute: async (params, ctx) => {
  const 分析结果 = await 分析输入(params, ctx);

  if (分析结果.类型 === '图片') {
    return await 处理图片(分析结果, ctx);
  } else if (分析结果.类型 === '文档') {
    return await 处理文档(分析结果, ctx);
  } else {
    return await 通用处理(分析结果, ctx);
  }
}

迭代型

对数据集合中的每个元素执行相同操作：

// 迭代型技能结构
execute: async (params, ctx) => {
  const items = await ctx.db.find('user_data', params.集合, {});
  const results = [];

  for (const item of items) {
    try {
      const result = await 处理单项(item, ctx);
      results.push({ id: item._id, status: 'success', result });
    } catch (error) {
      results.push({ id: item._id, status: 'failed', error: error.message });
      // 单项失败不中断整体
    }
  }

  return { 总数: items.length, 结果: results };
}

编排型

协调多个工具或其他扩展完成复合任务：

// 编排型技能结构
execute: async (params, ctx) => {
  // 并行获取多源数据
  const [用户数据, 历史记录, 外部数据] = await Promise.all([
    ctx.db.find('user_data', '用户', { id: params.userId }),
    ctx.db.find('user_data', '历史', { userId: params.userId }),
    ctx.fetch('https://api.example.com/data')
  ]);

  // 调用其他扩展的能力
  const 分析结果 = await ctx.extensions.call('数据分析', '综合分析', {
    数据: { 用户数据, 历史记录, 外部数据 }
  });

  // 生成报告
  const 报告 = await ctx.extensions.call('文档', '生成报告', {
    模板: '综合分析报告',
    数据: 分析结果
  });

  return 报告;
}

在 manifest 中注册

只需在 manifest.js 中指定 skills 目录路径，系统自动扫描注册：

// manifest.js
export default {
  name: '数据处理',
  // ...
  skills: './skills',   // 指向 skills 目录
  tools: './tools',     // 指向 tools 目录
  // ...
};

技能与策略的配合

技能通常由 strategy.js 中的策略层调度。策略决定「用哪个技能」，技能决定「怎么执行」：

// strategy.js
export default {
  // 策略根据意图选择技能
  plan: (intent, context) => {
    if (intent.action === '清洗数据') {
      return { skill: '数据清洗', params: intent.params };
    }
    if (intent.action === '生成报告') {
      return { skill: '报告生成', params: intent.params };
    }
    // 无匹配技能时，回退到工具直接调用
    return { useTools: true };
  }
};

完整示例：代码审查技能

export default {
  name: '代码审查',
  description: '对代码变更进行全面审查，检查规范性、安全性、性能',

  parameters: {
    type: 'object',
    properties: {
      文件列表: {
        type: 'array',
        items: { type: 'string' },
        description: '需要审查的文件路径列表'
      },
      审查维度: {
        type: 'array',
        items: {
          type: 'string',
          enum: ['规范性', '安全性', '性能', '可读性', '测试覆盖']
        },
        default: ['规范性', '安全性', '性能']
      }
    },
    required: ['文件列表']
  },

  usesTools: ['读取文件', '获取Git差异'],

  execute: async (params, ctx) => {
    const { 文件列表, 审查维度 = ['规范性', '安全性', '性能'] } = params;

    const 审查结果 = [];

    for (const 文件路径 of 文件列表) {
      // 读取文件内容
      const 内容 = await ctx.storage.read(文件路径);

      if (!内容) {
        审查结果.push({
          文件: 文件路径,
          状态: '跳过',
          原因: '文件不存在或为空'
        });
        continue;
      }

      // 每个维度独立检查
      const 问题列表 = [];

      for (const 维度 of 审查维度) {
        const 维度问题 = await 按维度检查(内容, 维度, ctx);
        问题列表.push(...维度问题);
      }

      审查结果.push({
        文件: 文件路径,
        状态: 问题列表.length > 0 ? '有问题' : '通过',
        问题数: 问题列表.length,
        问题: 问题列表
      });
    }

    const 总问题数 = 审查结果.reduce((sum, r) => sum + (r.问题数 || 0), 0);

    return {
      success: true,
      总结: `审查了 ${文件列表.length} 个文件，发现 ${总问题数} 个问题`,
      详情: 审查结果
    };
  }
};

// 辅助函数
async function 按维度检查(内容, 维度, ctx) {
  // 实际实现中这里会调用 LLM 或规则引擎
  // 此处为简化示例
  const 问题 = [];

  if (维度 === '安全性') {
    if (内容.includes('eval(')) {
      问题.push({ 维度, 严重度: '高', 描述: '检测到 eval() 调用，存在代码注入风险' });
    }
    if (内容.includes('password') && 内容.includes('console.log')) {
      问题.push({ 维度, 严重度: '高', 描述: '疑似在日志中输出密码' });
    }
  }

  if (维度 === '规范性') {
    if (内容.length > 600 * 80) {  // 粗略估算超过 600 行
      问题.push({ 维度, 严重度: '中', 描述: '文件超过 600 行，建议按职责拆分' });
    }
  }

  return 问题;
}

最佳实践

  • 故障隔离：处理列表数据时，单项失败不要中断整个技能，记录错误后继续
  • 进度反馈：长时间运行的技能应通过 ctx.log 输出进度
  • 可组合：技能之间可以互相调用（通过 ctx.extensions），设计时考虑复用
  • 幂等性：尽量让技能多次执行产生相同结果，避免重复执行导致数据异常
  • 超时处理：外部调用设置合理超时，避免无限等待
